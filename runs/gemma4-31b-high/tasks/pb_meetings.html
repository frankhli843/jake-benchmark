def _parse_dt(s):
"""Parse flexible datetime string."""
for fmt in (
"%Y-%m-%dT%H:%M:%S",
"%Y-%m-%dT%H:%M",
"%Y-%m-%d %H:%M:%S",
"%Y-%m-%d %H:%M",
"%Y-%m-%d",
):
try:
return datetime.strptime(s, fmt)
except ValueError:
continue
raise ValueError(f"Cannot parse date: {s}")

def _fmt_date(iso_str):
"""Format ISO string for display."""
try:
dt = _parse_dt(iso_str)
return dt.strftime("%Y-%m-%d %H:%M")
except Exception:
return iso_str

def _today():
return datetime.now().replace(hour=0, minute=0, second=0, microsecond=0)

def cmd_calendar_list(args):
events = _load("calendar")
if not events:
print("No events found.")
return

# Filter by date range
if args.from_date:
from_dt = _parse_dt(args.from_date)
events = [e for e in events if _parse_dt(e["start"]) >= from_dt]
else:
# Default: from now
events = [e for e in events if _parse_dt(e["start"]) >= _today()]

if args.to_date:
to_dt = _parse_dt(args.to_date)
events = [e for e in events if _parse_dt(e["start"]) <= to_dt]
else:
# Default: next 7 days
events = [e for e in events if _parse_dt(e["start"]) <= _today() + timedelta(days=7)]

events.sort(key=lambda e: e["start"])

if not events:
print("No events found in the specified range.")
return
