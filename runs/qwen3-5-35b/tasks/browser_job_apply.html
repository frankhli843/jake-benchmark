#!/usr/bin/env node
/**
 * chrome-mcp-client.mjs — Spawns chrome-devtools-mcp as a child process,
 * sends MCP JSON-RPC over stdio, extracts the tool call result, and exits.
 * 
 * Usage: node chrome-mcp-client.mjs <tool_name> [key=value ...]
 */
import { spawn } from 'node:child_process';
import { createInterface } from 'node:readline';

const MCP_MAIN = '/home/linuxbrew/.linuxbrew/lib/node_modules/chrome-devtools-mcp/build/src/main.js';
const CDP_PORT = process.env.CDP_PORT || '9222';

const toolName = process.argv[2];
if (!toolName) {
  console.error('Usage: node chrome-mcp-client.mjs <tool_name> [key=value ...]');
  process.exit(1);
}

// Parse key=value args into object
const args = {};
for (let i = 3; i < process.argv.length; i++) {
  const eq = process.argv[i].indexOf('=');
  if (eq === -1) continue;
  const key = process.argv[i].substring(0, eq);
  const val = process.argv[i].substring(eq + 1);
  // Keep all values as strings (MCP expects strings for uid, pageId, etc.)
  args[key] = val;
}

// Spawn the MCP server as a child process
const child = spawn(process.execPath, [MCP_MAIN, '--browserUrl', `http://127.0.0.1:${CDP_PORT}`], {
  stdio: ['pipe', 'pipe', 'pipe'],
  env: { ...process.env, CHROME_DEVTOOLS_MCP_NO_USAGE_STATISTICS: '1' },
});

// Read stdout line by line for JSON-RPC responses
const rl = createInterface({ input: child.stdout });
let gotInit = false;

rl.on('line', (line) => {
  let msg;
  try { msg = JSON.parse(line); } catch { return; }
  
  if (msg.id === 1 && !gotInit) {
    // Initialize response received — send initialized notification + tool call
    gotInit = true;
    const notify = JSON.stringify({jsonrpc:'2.0', method:'notifications/initialized'});
    const toolReq = JSON.stringify({jsonrpc:'2.0', id:2, method:'tools/call', params:{name: toolName, arguments: args}});
    child.stdin.write(notify + '\n');
    child.stdin.write(toolReq + '\n');
  }
  
  if (msg.id === 2) {
    // Tool call response
    if (msg.error) {
      console.error('[chrome-mcp] Error:', JSON.stringify(msg.error));
      child.kill();
      process.exit(1);
    }
    const content = msg.result?.content || [];
    for (const c of content) {
      if (c.type === 'text') console.log(c.text);
      else if (c.type === 'image') console.log(`[image: ${c.mimeType||'png'}, ${(c.data||'').length} bytes base64]`);
      else console.log(JSON.stringify(c));
    }
    child.kill();
    process.exit(0);
  }
});

// Handle stderr (suppress noisy MCP server logs)
child.stderr.on('data', () => {});

child.on('error', (err) => {
  console.error('[chrome-mcp] Failed to start MCP server:', err.message);
  process.exit(1);
});

// Send initialize request

  jsonrpc: '2.0', id: 1, method: 'initialize',
  params: {
    protocolVersion: '2024-11-05',
    capabilities: {},
    clientInfo: { name: 'chrome-mcp-call', version: '1.0.0' }
  }
});
child.stdin.write(initReq + '\n');

// Safety timeout
setTimeout(() => {
  console.error('[chrome-mcp] Timeout after 30s');
  child.kill();
  process.exit(1);
}, 30000);
